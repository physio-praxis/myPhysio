Table customer {
  customer_id integer [primary key]
  name varchar
  email varchar
  phone_number varchar
  contact_information varchar
  address varchar
}

Table pet {
  pet_id integer [primary key]
  name varchar
  species varchar
  breed varchar
  age integer
  medical_history varchar [note: 'Medical history of the pet']
  customer_id integer
}

Table appointment_status {
  status_id integer [primary key]
  status_name varchar
}

Table appointment {
  appointment_id integer [primary key]
  date_time timestamp
  status_id integer
  customer_id integer
  pet_id integer
}

Table pet_treatment {
  pet_treatment_id integer [primary key]
  pet_id integer
  treatment_id integer
}

Table treatment {
  treatment_id integer [primary key]
  name varchar
  description text [note: 'Description of the treatment']
}

Table invoice {
  invoice_id integer [primary key]
  date_issued timestamp
  amount decimal
  customer_id integer
}

Table payment {
  payment_id integer [primary key]
  amount_paid decimal
  payment_date timestamp
  invoice_id integer
}

Table pet_medical_issue {
  pet_medical_issue_id integer [primary key]
  pet_id integer
  medical_issue_id integer
}

Table medical_issue {
  medical_issue_id integer [primary key]
  medical_issue varchar
}

Ref: pet.customer_id > customer.customer_id // many-to-one
Ref: appointment.customer_id > customer.customer_id // many-to-one
Ref: appointment.pet_id > pet.pet_id // many-to-one
Ref: appointment.status_id > appointment_status.status_id // many-to-one
Ref: pet_treatment.pet_id > pet.pet_id // many-to-one
Ref: pet_treatment.treatment_id > treatment.treatment_id // many-to-one
Ref: invoice.customer_id > customer.customer_id // many-to-one
Ref: payment.invoice_id > invoice.invoice_id // many-to-one
Ref: pet_medical_issue.pet_id > pet.pet_id // many-to-one
Ref: pet_medical_issue.medical_issue_id > medical_issue.medical_issue_id // many-to-one
